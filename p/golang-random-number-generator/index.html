func init() {
		rand.Seed(time.Now().UnixNano())
}

var runes = 
[]rune("____ABCDEFGHIJKLMNOPQRSTUVWXYZ1234567890123456789012345678901234567890")

func GenerateAPIKey(n int) string {
		b := make([]rune, n)
		for i := range b {
				b[i] = runes[rand.Intn(len(runes))]
		}
return string(b)

}
